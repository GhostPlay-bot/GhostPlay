return (
        <div
            className='h-screen w-screen bg-cover bg-center flex items-center justify-center'
            style={{
                backgroundImage: "url('/bg-cover.png')",
            }}
        >
            <div className='bg-black/90 rounded-2xl shadow-xl px-8 py-10 max-w-sm w-full text-center'>
                <h1 className="text-2xl font-bold mb-6">Sign In</h1>

                <button
                    onClick={handleGoogleSignIn}
                    className="flex items-center justify-center w-full py-3 px-4 bg-white rounded-full shadow hover:cursor-pointer hover:shadow-lg transition"
                >
                    <img
                        src="https://www.svgrepo.com/show/355037/google.svg"
                        alt="Google"
                        className="w-5 h-5 mr-2"
                    />
                    <span className="text-gray-800 font-medium">
                        Continue with Google
                    </span>
                </button>
            </div>
        </div>
    )

const handleGoogleSignIn = async () => {
        const { error } = await supabase.auth.signInWithOAuth({
            provider: 'google',
            options: {
                redirectTo: 'http://localhost:5173'
            }
        })
        if (error) console.error('Google sign-in error:', error.message);
    }
